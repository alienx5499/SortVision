Selection Sort (in-place)
Time: Best/Avg/Worst O(n^2)
Space: O(1)

procedure selectionSort(array)
    n <- length(array)

    for i <- 0 to n - 2
        minIndex <- i

        for j <- i + 1 to n - 1
            if array[j] < array[minIndex]
                minIndex <- j
            end if
        end for

        if minIndex != i
            swap(array[i], array[minIndex])
        end if
    end for
end procedure
