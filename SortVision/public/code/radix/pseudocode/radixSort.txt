Radix Sort (LSD, base 10, non-negative integers)
Time: O(d * (n + k))
Space: O(n + k)

procedure radixSort(array)
    maxVal <- maximum(array)

    for exp <- 1 while maxVal / exp > 0
        countingSortByExp(array, exp)
        exp <- exp * 10
    end for
end procedure

procedure countingSortByExp(array, exp)
    output <- array of size n
    count <- array[0..9] initialized to 0

    for each value in array
        digit <- (value / exp) % 10
        count[digit] <- count[digit] + 1
    end for

    prefix-sum count

    for i <- n - 1 down to 0
        digit <- (array[i] / exp) % 10
        output[count[digit] - 1] <- array[i]
        count[digit] <- count[digit] - 1
    end for

    copy output back to array
end procedure
