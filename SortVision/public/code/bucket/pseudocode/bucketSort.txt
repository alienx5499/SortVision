Bucket Sort (for numbers in [0, 1))
Time: Avg O(n + k), Worst O(n^2)
Space: O(n + k)

procedure bucketSort(array)
    n <- length(array)
    buckets <- n empty lists

    for each value in array
        idx <- floor(value * n)
        if idx = n
            idx <- n - 1
        end if
        append value to buckets[idx]
    end for

    for each bucket in buckets
        sort(bucket)
    end for

    concatenate buckets back into array
end procedure
