Heap Sort (in-place, max-heap)
Time: Best/Avg/Worst O(n log n)
Space: O(1)

procedure heapSort(array)
    n <- length(array)

    for i <- floor(n / 2) - 1 down to 0
        heapify(array, n, i)
    end for

    for i <- n - 1 down to 1
        swap(array[0], array[i])
        heapify(array, i, 0)
    end for
end procedure

procedure heapify(array, n, i)
    largest <- i
    left <- 2 * i + 1
    right <- 2 * i + 2

    if left < n and array[left] > array[largest]
        largest <- left
    end if

    if right < n and array[right] > array[largest]
        largest <- right
    end if

    if largest != i
        swap(array[i], array[largest])
        heapify(array, n, largest)
    end if
end procedure
